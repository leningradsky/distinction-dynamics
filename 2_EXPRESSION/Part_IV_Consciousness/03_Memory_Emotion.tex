% Memory and Emotion - Part IV
\chapter{Memory, Emotion, and Temporal Experience}\label{ch:memory-emotion}
\epigraph{The past is present in distinction.}{---}

\section{Memory as Distinction Trace}

\[
H(t) = \int_0^t K(t-\tau) \Delta(\tau) \, d\tau
\]

Where $K$ is the forgetting kernel (typically exponential).

\section{Subjective Time}

Internal time = $\|\Delta(t) - H(t)\|$

\begin{itemize}
    \item If $\Delta \approx H$: stagnation, ``time stands still''
    \item If $\Delta \neq H$: flow, ``time passes''
\end{itemize}

\section{Emotion as Rate}

$E(t) = \|d\Delta/dt\|$

Emotion is the \emph{speed} of distinction change, not its content.
