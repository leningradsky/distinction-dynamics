\section{Consistency Conditions}

\subsection{Condition 1: Rank $\geq 2$}

From triadic necessity:
\begin{equation}
\mathrm{rank}(G) \geq 2
\end{equation}

This eliminates $SU(2)$, $U(1)$, and all rank-1 groups.

\subsection{Condition 2: Anomaly Freedom}

\begin{definition}[Gauge Anomaly]
A gauge theory has an \textbf{anomaly} if quantum corrections break gauge invariance. This renders the theory inconsistent.
\end{definition}

\begin{theorem}[Anomaly Cancellation Requirement]
For a consistent quantum gauge theory:
\begin{equation}
\mathcal{A} = \sum_f Q_f^3 = 0
\end{equation}
where the sum runs over all fermion representations.
\end{theorem}

\subsubsection{SU(3) Anomaly Check}

For $SU(3)$ with fundamental representation $\mathbf{3}$:
\begin{itemize}
    \item Quarks: 3 colors $\times$ 3 generations = 9 states
    \item Antiquarks: 3 colors $\times$ 3 generations = 9 states
    \item Anomaly: $\mathcal{A}(\mathbf{3}) + \mathcal{A}(\bar{\mathbf{3}}) = 0$ ✓
\end{itemize}

\textbf{SU(3) is anomaly-free.}

\subsubsection{SU(4) Anomaly Check}

For $SU(4)$ with 3 generations:
\begin{itemize}
    \item Fundamental $\mathbf{4}$: 4 states per generation
    \item With Standard Model matter content: anomalous
    \item Would require exactly 4 generations for cancellation
\end{itemize}

\textbf{SU(4) requires fine-tuning of generations.}

\subsection{Condition 3: Asymptotic Freedom}

\begin{definition}[Asymptotic Freedom]
A gauge theory is \textbf{asymptotically free} if the coupling constant decreases at high energies:
\begin{equation}
\beta(g) = \mu \frac{dg}{d\mu} < 0 \quad \text{for small } g
\end{equation}
\end{definition}

\begin{theorem}[Gross-Wilczek-Politzer, 1973]
Non-abelian gauge theories are asymptotically free if:
\begin{equation}
N_f < \frac{11}{2} N_c
\end{equation}
where $N_f$ = number of quark flavors, $N_c$ = number of colors.
\end{theorem}

For $SU(3)$ with $N_c = 3$:
\begin{equation}
N_f < \frac{11}{2} \times 3 = 16.5
\end{equation}

With 6 flavors (u, d, s, c, b, t): $6 < 16.5$ ✓

\textbf{SU(3) is asymptotically free.}

For $SU(N)$ with $N > 3$: requires more flavors to match observations, may lose asymptotic freedom.

\subsection{Condition 4: Confinement}

\begin{definition}[Confinement]
A gauge theory is \textbf{confining} if colored states cannot be isolated; only color-singlets are physical.
\end{definition}

\begin{theorem}[Confinement Criterion]
Asymptotically free theories with sufficient coupling at low energies exhibit confinement.
\end{theorem}

For $SU(3)$:
\begin{itemize}
    \item Strong coupling at low energy: $\alpha_s(M_Z) \approx 0.118$
    \item Confinement scale: $\Lambda_{\text{QCD}} \approx 200$ MeV
    \item Hadrons (baryons, mesons) are observed
\end{itemize}

\textbf{SU(3) confines.}

\subsection{Condition 5: det = 1 (Special Unitary)}

\begin{theorem}[Unitarity + Tracelessness]
\begin{equation}
SU(N) = \{U \in GL(N, \mathbb{C}) : U^\dagger U = I, \det U = 1\}
\end{equation}
\end{theorem}

Why $\det = 1$?

\begin{proof}[Argument from quantum consistency]
\begin{enumerate}
    \item Gauge transformations must preserve probability: $U^\dagger U = I$ (unitarity).
    \item The overall phase $e^{i\theta}$ is a global symmetry, not gauge.
    \item Gauging $U(1)$ phase separately leads to $U(N) = SU(N) \times U(1)$.
    \item The $U(1)$ factor has gravitational anomalies unless canceled.
    \item For color: no such $U(1)$ is observed $\Rightarrow$ $\det = 1$.
\end{enumerate}
\end{proof}

\textbf{The "S" in SU(3) is required by consistency.}
