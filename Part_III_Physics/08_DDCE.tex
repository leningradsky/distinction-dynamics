\section{Problem Statement}

Standard cosmology assumes the accelerated expansion of the universe is driven by dark energy or the cosmological constant. These approaches treat physical law and informational structure as fixed, timeless features.

We introduce a competing, testable hypothesis:

\begin{center}
\fbox{\parbox{0.9\textwidth}{
\textbf{DDCE Hypothesis}: The expansion of spacetime is a dynamic response to the evolving capacity of the universe to generate, store, and differentiate information.
}}
\end{center}

Rather than treating geometry as a passive background, we propose a coupling between the production of distinctions (informational complexity) and the growth of spatial volume.

\section{Definitions}

\begin{definition}[Distinction ($\Delta$)]
A stable, recognizable difference between physical or informational states.
\end{definition}

\begin{definition}[Reaction Function ($F$)]
A law of nature: the mapping from a configuration of distinctions to predicted future distinctions.
\end{definition}

\begin{definition}[Meta-Reaction ($M$)]
A process which modifies the reaction function $F$ over time. In physics, $M$ corresponds to:
\begin{itemize}
    \item Renormalization group flow
    \item Symmetry breaking
    \item Running coupling constants
    \item Effective field dynamics
\end{itemize}
\end{definition}

\begin{definition}[Distinction Complexity ($\mathcal{C}$)]
A measure of the total number of stable, recursive, representation-independent ways a system can generate and respond to distinctions.
\end{definition}

\section{Core Claim}

\begin{hypothesis}[DDCE]
\begin{equation}
\frac{dV}{dt} \propto \frac{d\mathcal{C}}{dt}
\end{equation}
where $V$ is cosmological spacetime volume and $\mathcal{C}$ is distinction complexity.
\end{hypothesis}

The universe expands in proportion to the rate at which new distinctions (new ways of classifying and responding to states) are generated within it.

\section{Rationale}

\subsection{Physical Entropification (Bottom-up)}

Physics produces distinctions passively:
\begin{itemize}
    \item Thermodynamic entropy
    \item Quantum decoherence
    \item Phase transitions
\end{itemize}

This production is slow, bounded, and often polynomial in system size.

\subsection{Biological Adaptation (Active)}

Life produces distinctions faster:
\begin{itemize}
    \item Immune repertoires: $\sim 10^{11}$ configurations
    \item Genetic diversification
    \item Neural plasticity
\end{itemize}

This production is exponential.

\subsection{Cognitive Recursion (Super-exponential)}

Mind produces distinctions of distinctions:
\begin{itemize}
    \item Language: $\sim 10^{3000}$ possible sentences
    \item Mathematics
    \item Abstract models
    \item Scientific theories
\end{itemize}

This production is super-exponential, because every new $M$ allows new $F$, which allows new $\Delta$, recursively.

Thus:
\begin{equation}
\left.\frac{d\mathcal{C}}{dt}\right|_{\text{mind}} \gg \left.\frac{d\mathcal{C}}{dt}\right|_{\text{life}} \gg \left.\frac{d\mathcal{C}}{dt}\right|_{\text{physics}}
\end{equation}

The cosmological expansion is therefore \textbf{reactive}, not primitive.

\section{Formal Dynamical System}

\subsection{Variables}

\begin{align}
\Delta(t) &\in \mathbb{R}^+ && \text{(distinctions)} \\
F(t) &\in \mathbb{R}^+ && \text{(reaction laws)} \\
M(t) &\in \mathbb{R}^+ && \text{(meta-reactions)} \\
V(t) &\in \mathbb{R}^+ && \text{(cosmic volume)}
\end{align}

\subsection{Dynamical Equations}

\begin{align}
\frac{d\Delta}{dt} &= a + bF + cM \label{eq:delta-dyn} \\
\frac{dF}{dt} &= \beta(M, \Delta) \label{eq:F-dyn} \\
\frac{dM}{dt} &= \gamma(\Delta, F) \label{eq:M-dyn} \\
\frac{dV}{dt} &= k(\Delta + F + M) \label{eq:V-dyn}
\end{align}

Where $c \gg b \gg a$ (mind generates distinctions faster than life, life faster than physics).

\subsection{$\beta$-Function (DD-Version of RG)}

\begin{equation}
\beta(M, \Delta) = \alpha_1 \Delta - \alpha_2 F + \alpha_3 M
\end{equation}

\begin{itemize}
    \item $\alpha_1 > 0$: new distinctions require new laws
    \item $\alpha_2 > 0$: ``fatigue'' or stabilization of laws
    \item $\alpha_3 > 0$: mind creates new law models
\end{itemize}

\subsection{$\gamma$-Function}

\begin{equation}
\gamma(\Delta, F) = \beta_1 \Delta - \beta_2 M + \beta_3 F
\end{equation}

\begin{itemize}
    \item $\beta_1$: distinctions stimulate meta-distinctions
    \item $\beta_2$: fatigue / inertia
    \item $\beta_3$: laws stimulate revision of laws
\end{itemize}

\subsection{Key Result: Mutual Activation}

The system has cross-activation:
\begin{align}
\frac{dF}{dt} &\sim \alpha_1 \Delta + \alpha_3 M \\
\frac{dM}{dt} &\sim \beta_1 \Delta + \beta_3 F
\end{align}

This means: \textbf{$F$ and $M$ exponentially amplify each other}.

Defining $\lambda = \alpha_1 \beta_1 + \alpha_3 \beta_3 > 0$:
\begin{equation}
F(t) \sim e^{\sqrt{\lambda} t}, \quad M(t) \sim e^{\sqrt{\lambda} t}
\end{equation}

Therefore:
\begin{equation}
V(t) \sim \frac{1}{\sqrt{\lambda}} e^{\sqrt{\lambda} t}
\end{equation}

\textbf{$V(t)$ grows exponentially} — exactly what we see in data about dark energy and accelerating universe.

\section{Lagrangian Formulation}

\begin{equation}
L = \frac{1}{2}\left(\dot{\Delta}^2 + \dot{F}^2 + \dot{M}^2 + \dot{V}^2\right) - U(\Delta, F, M, V)
\end{equation}

where the potential:
\begin{equation}
U = -\alpha_1 \Delta F - \alpha_3 F M - \beta_1 \Delta M - k(\Delta + F + M)V
\end{equation}

Euler-Lagrange equations yield:
\begin{align}
\ddot{\Delta} &= \alpha_1 F + \beta_1 M + kV \\
\ddot{F} &= \alpha_1 \Delta + \alpha_3 M + kV \\
\ddot{M} &= \beta_1 \Delta + \alpha_3 F + kV \\
\ddot{V} &= k(\Delta + F + M)
\end{align}

\textbf{Key effect}: All variables accelerate, not just grow.

\begin{equation}
\ddot{V} > 0 \quad \text{if} \quad \Delta, F, M > 0
\end{equation}

This is the \textbf{accelerating expansion of the cosmos} — driven by distinction complexity.
